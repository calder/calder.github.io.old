00324         // Get a character, while interpreting entities.
00325         // The length can be from 0 to 4 bytes.
00326         inline static const char* GetChar( const char* p, char* _value, int* length, TiXmlEncoding encoding )
00327         {
00328                 assert( p );
00329                 if ( encoding == TIXML_ENCODING_UTF8 )
00330                 {
00331                         *length = utf8ByteTable[ *((const unsigned char*)p) ];
00332                         assert( *length >= 0 && *length < 5 );
00333                 }
00334                 else
00335                 {
00336                         *length = 1;
00337                 }
00338 
00339                 if ( *length == 1 )
00340                 {
00341                         if ( *p == '&' )
00342                                 return GetEntity( p, _value, length, encoding );
00343                         *_value = *p;
00344                         return p+1;
00345                 }
00346                 else if ( *length )
00347                 {
00348                         //strncpy( _value, p, *length );        // lots of compilers don't like this function (unsafe),
00349                                                                                                 // and the null terminator isn't needed
00350                         for( int i=0; p[i] && i<*length; ++i ) {
00351                                 _value[i] = p[i];
00352                         }
00353                         return p + (*length);
00354                 }
00355                 else
00356                 {
00357                         // Not valid text.
00358                         return 0;
00359                 }
00360         }
