00341     {
00342         // Check that source exists
00343         if (!hasChildData(group,name)) { throw Exception("AC::SerialData: moveChild cannot copy non-existent child \""+name+"\" in group \""+group+"\"!"); }

00345         if (hasChildData(newGroup,newName)) { throw Exception("AC::SerialData: moveChild cannot overwrite existing child \""+newName+"\" in group \""+newGroup+"\"!"); }
00346         //-------------------------------------------------
00347         // Move child or child data, whichever exists

00349         {
00350             mChildren[newGroup][newName] = mChildren[group][name];

00354             if (mChildren[group].size() == 0) { mChildren.erase(group); }

00356         else
00357         {
00358             mChildrenData[newGroup][newName] = mChildrenData[group][name];
00359             mChildrenData[newGroup][newName].setName(newName);
00360             mChildrenData[group].erase(name);

00617         {

00619         SerialData& self = *const_cast<SerialData*>(this);
00620         //-------------------------------------------------
00621         // Setup variables
00622         TiXmlElement* root = new TiXmlElement(mType);
00623         if (hasName()) { root->SetAttribute("name", mName); }
00624         TiXmlElement* elem;
00625         TiXmlText* text;
00626         //--------------------------------------------------
00627         // Synchronize mStringData with mTypedData
00628         for (TypedData::iterator it = self.mTypedData.begin(); it != mTypedData.end(); ++it)
00629         {
00630             self.mStringData[it->first] = PrimitiveString(it->second->toString(), it->second->getType());
00631         }
00632         //--------------------------------------------------
00633         // Write out primitives
00634         for (StringData::iterator it = self.mStringData.begin(); it != mStringData.end(); ++it)
00635         {
00636             elem = new TiXmlElement(it->first);
00637             text = new TiXmlText(it->second);
00638             if (includeTypes && it->second.type() != "none") { elem->SetAttribute("type", it->second.type()); }
00639             elem->LinkEndChild(text);
00640             root->LinkEndChild(elem);
00641         }
00642         //--------------------------------------------------
00643         // Write out children data
00644         map<string, TiXmlElement*> groupNodes;
00645         TiXmlElement* curGroupNode;
00646         // Write out pure-data children
00647         for (ChildrenData::const_iterator it=mChildrenData.begin(); it!=mChildrenData.end(); ++it)
00648         {
